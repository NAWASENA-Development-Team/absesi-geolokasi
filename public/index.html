/* ============================================================ KONFIGURASI LOKASI SEKOLAH — UBAH SESUAI KEBUTUHAN ============================================================ */ const SCHOOL_LOCATION = { latitude: -6.469821658683405, // ← Ganti jika perlu longitude: 107.07387463910511, // ← Ganti jika perlu }; const RADIUS_METER = 150; // ← Radius valid dalam meter /* ============================================================ JAM LANGSUNG (LIVE CLOCK) ============================================================ */ const HARI = ['Minggu','Senin','Selasa','Rabu','Kamis','Jumat','Sabtu']; const BULAN = ['Januari','Februari','Maret','April','Mei','Juni', 'Juli','Agustus','September','Oktober','November','Desember']; function getOffsetLabel() { // getTimezoneOffset() mengembalikan nilai NEGATIF untuk zona timur (misal WIB = -420) const offsetMin = -new Date().getTimezoneOffset(); const sign = offsetMin >= 0 ? '+' : '-'; const abs = Math.abs(offsetMin); const h = String(Math.floor(abs / 60)).padStart(2, '0'); const m = String(abs % 60).padStart(2, '0'); return `UTC${sign}${h}:${m}`; } function pad(n) { return String(n).padStart(2, '0'); } function updateClock() { const now = new Date(); // ── JAM format 24 jam (HH:MM:SS) ────────────────────────────────── document.getElementById('clock').textContent = `${pad(now.getHours())}:${pad(now.getMinutes())}:${pad(now.getSeconds())}`; // ── TANGGAL dalam Bahasa Indonesia ───────────────────────────────── const namaHari = HARI[now.getDay()]; const tanggal = now.getDate(); const namaBulan = BULAN[now.getMonth()]; const tahun = now.getFullYear(); const offsetLabel = getOffsetLabel(); document.getElementById('date').textContent = `${namaHari}, ${tanggal} ${namaBulan} ${tahun} • ${offsetLabel}`; } updateClock(); // panggil sekali langsung setInterval(updateClock, 1000); // update tiap detik /* ============================================================ ORG TOGGLE ============================================================ */ function selectOrg(el) { document.querySelectorAll('.org-btn').forEach(b => b.classList.remove('active')); const kelas = document.getElementById('kelas').value; const org = document.getElementById('organisasi').value; if (!nama) { showFormError('Nama lengkap harus diisi!'); return false; } if (!kelas) { showFormError('Kelas harus dipilih!'); return false; } if (!org) { showFormError('Pilih organisasi terlebih dahulu!'); return false; } return true; } function showFormError(msg) { // Tampilkan notif error kecil (non-blocking) const toast = document.createElement('div'); document.body.appendChild(toast); setTimeout(() => toast.remove(), 3000); } /* ============================================================ RUMUS HAVERSINE — Menghitung jarak dua koordinat (meter) Referensi: https://en.wikipedia.org/wiki/Haversine_formula ============================================================ */ function hitungJarak(lat1, lon1, lat2, lon2) { const R = 6371000; // Radius Bumi dalam meter const phi1 = (lat1 * Math.PI) / 180; const phi2 = (lat2 * Math.PI) / 180; const dPhi = ((lat2 - lat1) * Math.PI) / 180; const dLambda = ((lon2 - lon1) * Math.PI) / 180; const a = Math.sin(dPhi / 2) * Math.sin(dPhi / 2) + Math.cos(phi1) * Math.cos(phi2) * Math.sin(dLambda / 2) * Math.sin(dLambda / 2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); return R * c; // hasil dalam meter } /* ============================================================ FUNGSI UTAMA — HANDLE ABSEN ============================================================ */ async function handleAbsen() { // 1. Validasi form terlebih dahulu if (!validateForm()) return; // 2. Hitung jarak dari sekolah menggunakan Haversine const jarak = hitungJarak( userLat, userLon, SCHOOL_LOCATION.latitude, SCHOOL_LOCATION.longitude ); const jarakRounded = Math.round(jarak); // Update badge jarak const distBadge = document.getElementById('loc-dist'); distBadge.style.display = 'block'; distBadge.textContent = `${jarakRounded} m`; // 6. ──────── VALIDASI KRITIS: CEK RADIUS ──────── if (jarak > RADIUS_METER) { // ❌ DI LUAR SEKOLAH — TOLAK, JANGAN KIRIM KE DATABASE setLoading(false); tampilkanHasil( 'error', '🚫', 'Di Luar Area Sekolah', `Absensi ditolak. Kamu terdeteksi sejauh ${jarakRounded} meter dari SMAN 2 Jonggol. Batas maksimal adalah ${RADIUS_METER} meter.`, { 'Koordinat Kamu': `${userLat.toFixed(6)}, ${userLon.toFixed(6)}`, 'Jarak Terdeteksi': `${jarakRounded} meter`, 'Radius Valid': `${RADIUS_METER} meter`, 'Status': '🔴 DITOLAK — Data tidak disimpan', } ); return; // ← BERHENTI DI SINI, tidak ada data yang dikirim } // ✅ DALAM RADIUS — Lanjutkan simpan ke Supabase const waktuAbsen = new Date().toISOString(); const namaInput = document.getElementById('nama').value.trim(); const kelasInput = document.getElementById('kelas').value; const divisiInput= document.getElementById('divisi').value.trim(); const orgInput = document.getElementById('organisasi').value; // 7. Kirim ke Vercel API Route → Neon PostgreSQL const response = await fetch('/api/absen', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nama: namaInput, kelas: kelasInput, divisi: divisiInput || null, organisasi: orgInput, latitude, longitude, jarak_meter: parseFloat(jarak.toFixed(2)), }), }); const result = await response.json(); setLoading(false); if (!response.ok) { tampilkanHasil( 'error', '⚠️', 'Gagal Menyimpan Data', 'Terjadi kesalahan: ' + (result.error || 'Unknown error') ); return; } // Tampilkan sukses (sama seperti sebelumnya) tampilkanHasil( 'success', '✅', 'Absensi Berhasil!', `Selamat datang, ${namaInput}! Kehadiran kamu telah tercatat.`, { 'Nama' : namaInput, 'Kelas' : kelasInput, 'Organisasi': orgInput, 'Waktu' : new Date().toLocaleString('id-ID'), 'Jarak' : `${Math.round(jarak)} meter dari sekolah`, 'Status' : '🟢 HADIR — Tersimpan di database', } ); /* ============================================================ HELPER FUNCTIONS ============================================================ */ function setLoading(state) { const btn = document.getElementById('btn-absen'); btn.disabled = state; btn.classList.toggle('loading', state); } function updateLocStatus(icon, status, coords) { document.querySelector('.loc-icon').textContent = icon; document.getElementById('loc-status').textContent = status; document.getElementById('loc-coords').textContent = coords; } function tampilkanHasil(type, icon, title, desc, meta = null) { const resultCard = document.getElementById('result-card'); const formCard = document.getElementById('form-card'); document.getElementById('result-icon').textContent = icon; document.getElementById('result-title').textContent = title; document.getElementById('result-desc').innerHTML = desc; // Meta info const metaEl = document.getElementById('result-meta'); if (meta) { metaEl.innerHTML = Object.entries(meta) .map(([k, v]) => `
